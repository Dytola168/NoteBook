public class Phone {
   public String IME;
   public String Model;
    float Price;
   public String Color;
   public Phone(){
   }
   public Phone(String IME,String Model,float Price,String Color){
       this.IME = IME;
       this.Model= Model;
       this.Price = Price;
       this.Color =Color;
   }
   public void showData(){
       System.out.println("IME :"+IME);
       System.out.println("MOdel :"+Model);
       System.out.println("Price :"+Price);
       System.out.println("Color :"+Color);
   
   }
    
}
